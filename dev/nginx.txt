停止nginx
ps -ef | grep nginx		//在进程列表里 面找master进程，它的编号就是主进程号了。
kill -QUIT 主进程号		//从容停止Nginx
kill -TERM 主进程号		//快速停止Nginx
pkill -9 nginx		//强制停止Nginx

需要在putty/xxxx/nginx目录来执行文件
sudo ./sbin/nginx -c conf/nginx.conf	//有配置文件时用-c
sudo ./sbin/nginx
